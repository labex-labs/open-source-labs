# structs

Аналогично, `struct` можно деструктурировать, как показано ниже:

```rust
fn main() {
    struct Foo {
        x: (u32, u32),
        y: u32,
    }

    // Попробуйте изменить значения в структуре, чтобы увидеть, что произойдет
    let foo = Foo { x: (1, 2), y: 3 };

    match foo {
        Foo { x: (1, b), y } => println!("Первый элемент x равен 1, b = {},  y = {} ", b, y),

        // вы можете деструктурировать structs и переименовать переменные,
        // порядок не имеет значения
        Foo { y: 2, x: i } => println!("y равен 2, i = {:?}", i),

        // и вы также можете игнорировать некоторые переменные:
        Foo { y,.. } => println!("y = {}, нас не интересует x", y),
        // это вызовет ошибку: шаблон не упоминает поле `x`
        //Foo { y } => println!("y = {}", y),
    }

    let faa = Foo { x: (1, 2), y: 3 };

    // Вам не нужно использовать блок match для деструктуризации structs:
    let Foo { x : x0, y: y0 } = faa;
    println!("Вне блока: x0 = {x0:?}, y0 = {y0}");
}
```
