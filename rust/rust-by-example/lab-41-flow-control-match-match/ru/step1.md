# match

Rust предоставляет сопоставление шаблонов с помощью ключевого слова `match`, которое можно использовать подобно оператору `switch` в C. Первый подходящий вариант оценивается, и все возможные значения должны быть охвачены.

```rust
fn main() {
    let number = 13;
    // TODO ^ Попробуйте разные значения для `number`

    println!("Расскажи мне про {}", number);
    match number {
        // Сопоставьте с одним значением
        1 => println!("Один!"),
        // Сопоставьте с несколькими значениями
        2 | 3 | 5 | 7 | 11 => println!("Это простое число"),
        // TODO ^ Попробуйте добавить 13 в список простых чисел
        // Сопоставьте с включительным диапазоном
        13..=19 => println!("Подросток"),
        // Обработайте остальные случаи
        _ => println!("Не особый"),
        // TODO ^ Попробуйте закомментировать этот万用ный вариант
    }

    let boolean = true;
    // Match также является выражением
    let binary = match boolean {
        // Варианты match должны охватывать все возможные значения
        false => 0,
        true => 1,
        // TODO ^ Попробуйте закомментировать один из этих вариантов
    };

    println!("{} -> {}", boolean, binary);
}
```
