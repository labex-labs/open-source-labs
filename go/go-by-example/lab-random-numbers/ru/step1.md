# Случайные числа

Вам необходимо написать программу, которая генерирует случайные целые числа и числа с плавающей точкой в заданном диапазоне. Программа также должна быть capazна генерировать различные последовательности чисел путём изменения зерна.

- Программа должна использовать пакет `math/rand` для генерации случайных чисел.
- Программа должна генерировать случайные целые числа в заданном диапазоне.
- Программа должна генерировать случайные числа с плавающей точкой в заданном диапазоне.
- Программа должна быть capazна генерировать различные последовательности чисел путём изменения зерна.

```sh
# В зависимости от того, где вы запускаете этот пример,
# некоторые из сгенерированных чисел могут отличаться.
# Обратите внимание, что на Go Playground при
# инициализации зерна с использованием `time.Now()`
# по-прежнему получаются детерминированные результаты
# из-за особенностей реализации Playground.
$ go run random-numbers.go
81,87
0.6645600532184904
7.123187485356329,8.434115364335547
0,28
5,87
5,87

# См. документацию пакета [`math/rand`](https://pkg.go.dev/math/rand)
# для справки по другим случайным величинам, которые
# может предоставить Go.
```

Ниже представлен полный код:

```go
// В пакете `math/rand` Go реализована
// [генерация псевдослучайных чисел](https://en.wikipedia.org/wiki/Pseudorandom_number_generator).

package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {

	// Например, `rand.Intn` возвращает случайное `int` n,
	// `0 <= n < 100`.
	fmt.Print(rand.Intn(100), ",")
	fmt.Print(rand.Intn(100))
	fmt.Println()

	// `rand.Float64` возвращает `float64` `f`,
	// `0.0 <= f < 1.0`.
	fmt.Println(rand.Float64())

	// Это можно использовать для генерации случайных
	// чисел с плавающей точкой в других диапазонах,
	// например `5.0 <= f' < 10.0`.
	fmt.Print((rand.Float64()*5)+5, ",")
	fmt.Print((rand.Float64() * 5) + 5)
	fmt.Println()

	// По умолчанию генератор случайных чисел детерминирован,
	// поэтому по умолчанию он будет генерировать одну и ту же
	// последовательность чисел каждый раз. Чтобы получить
	// различные последовательности, задайте для него зерно,
	// которое будет меняться. Обратите внимание, что это не
	// безопасно использовать для случайных чисел, которые
	// должны быть секретными; для таких целей используйте
	// `crypto/rand`.
	s1 := rand.NewSource(time.Now().UnixNano())
	r1 := rand.New(s1)

	// Вызовите полученный `rand.Rand` точно так же, как
	// функции в пакете `rand`.
	fmt.Print(r1.Intn(100), ",")
	fmt.Print(r1.Intn(100))
	fmt.Println()

	// Если вы инициализируете источник одним и тем же числом,
	// он генерирует одну и ту же последовательность случайных
	// чисел.
	s2 := rand.NewSource(42)
	r2 := rand.New(s2)
	fmt.Print(r2.Intn(100), ",")
	fmt.Print(r2.Intn(100))
	fmt.Println()
	s3 := rand.NewSource(42)
	r3 := rand.New(s3)
	fmt.Print(r3.Intn(100), ",")
	fmt.Print(r3.Intn(100))
}

```
