# Парсинг URL-адресов

В лабораторном задании требуется разобрать пример URL-адреса, который включает в себя схему, информацию об аутентификации, хост, порт, путь, параметры запроса и фрагмент запроса. Разобранный URL-адрес должен использоваться для извлечения отдельных компонентов URL-адреса.

- Необходимо импортировать пакеты `url` и `net`.
- Пример URL-адрес должен быть разобран и проверен на ошибки.
- Схема, информация об аутентификации, хост, порт, путь, параметры запроса и фрагмент запроса должны быть извлечены из разобранного URL-адреса.
- Функция `SplitHostPort` должна быть использована для извлечения имени хоста и порта из поля `Host`.
- Функция `ParseQuery` должна быть использована для разбора параметров запроса в мапу.

```sh
# Запуск нашей программы по парсингу URL-адресов показывает все разные
# части, которые мы извлекли.
$ go run url-parsing.go
postgres
user:pass
user
pass
host.com:5432
host.com
5432
/path
f
k=v
map[k:[v]]
v

```

Ниже представлен полный код:

```go
// URL-адреса предоставляют [единый способ определения ресурсов](https://adam.herokuapp.com/past/2010/3/30/urls_are_the_uniform_way_to_locate_resources/).
// Вот, как разбирать URL-адреса на Go.

package main

import (
	"fmt"
	"net"
	"net/url"
)

func main() {

	// Мы будем разбирать этот пример URL-адреса, который включает в себя
	// схему, информацию об аутентификации, хост, порт, путь,
	// параметры запроса и фрагмент запроса.
	s := "postgres://user:pass@host.com:5432/path?k=v#f"

	// Разберите URL-адрес и убедитесь, что нет ошибок.
	u, err := url.Parse(s)
	if err!= nil {
		panic(err)
	}

	// Доступ к схеме простой.
	fmt.Println(u.Scheme)

	// `User` содержит всю информацию об аутентификации; вызовите
	// `Username` и `Password` для получения отдельных
	// значений.
	fmt.Println(u.User)
	fmt.Println(u.User.Username())
	p, _ := u.User.Password()
	fmt.Println(p)

	// `Host` содержит как имя хоста, так и порт,
	// если они есть. Используйте `SplitHostPort`, чтобы извлечь их.
	fmt.Println(u.Host)
	host, port, _ := net.SplitHostPort(u.Host)
	fmt.Println(host)
	fmt.Println(port)

	// Здесь мы извлекаем `path` и фрагмент после
	// `#`.
	fmt.Println(u.Path)
	fmt.Println(u.Fragment)

	// Чтобы получить параметры запроса в формате `k=v`,
	// используйте `RawQuery`. Вы также можете разобрать параметры запроса
	// в мапу. Разобранные мапы параметров запросов - это от
	// строк к средам строк, поэтому индексируйте `[0]`,
	// если вы хотите только первое значение.
	fmt.Println(u.RawQuery)
	m, _ := url.ParseQuery(u.RawQuery)
	fmt.Println(m)
	fmt.Println(m["k"][0])
}

```
