ax.set_xlim(np.datetime64('1989-06-01'), np.datetime64('2023-01-01'))

ax.tick_params(axis='both', which='both', labelsize='large',
               bottom=False, top=False, labelbottom=True,
               left=False, right=False, labelleft=True)
